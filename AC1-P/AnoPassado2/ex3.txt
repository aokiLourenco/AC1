# typedef struct
# { Alignment Size Offset
# char smp[10]; xx xx xx
# double av; xx xx xx
# int ns; xx xx xx
# char id; xx xx xx
# int sum; xx xx xx
# } t_cell; xx xx
double prcells(t_cell *tc, int size)
{
 double res, aux;
 int i;
 res = 0.0;
 for (i = 0; i < size; i++) {
 aux = (double)tc[i].ns / 3.597;
 res = res + aux;
 tc[i].av = aux;
 tc[i].sum = (int)res;
 }
 return res;
}
